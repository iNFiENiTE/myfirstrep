cd "<path>" - change directory
git ls-files - show list of file in the directory
git clone <url> - clone a repository

git add <filename> - add file that will be commit
git commit -m "message/note" - save the current state of the repository
git commit -am "message/note" - save all changed file and message [shorthand for add and commit]

git status - tell about git
git push - push the local changes to github
git pull - pull changes on github to local
